#include <bits/stdc++.h>
using namespace std;

int dfs(int i, int j, vector<vector<int>>& grid, vector<vector<int>>& dp, vector<vector<bool>>& vis) {
    if (i < 0 || j < 0 || i >= n || j >= m || grid[i][j] == 1)
        return INF;

    if (i == n - 1 && j == m - 1)
        return 0;

    if (vis[i][j])
        return INF;

    if (dp[i][j] != -1)
        return dp[i][j];

    vis[i][j] = true;
    int ans = INF;

    for (auto& d : dirs) {
        int ni = i + d[0];
        int nj = j + d[1];
        ans = min(ans, 1 + dfs(ni, nj, grid, dp, vis));
    }

    vis[i][j] = false;
    return dp[i][j] = ans;
}

int shortestPath(vector<vector<int>>& grid) {
    n = grid.size();
    m = grid[0].size();

    if (grid[0][0] == 1 || grid[n-1][m-1] == 1)
        return -1;

    vector<vector<int>> dp(n, vector<int>(m, -1));
    vector<vector<bool>> vis(n, vector<bool>(m, false));

    int result = dfs(0, 0, grid, dp, vis);
    return (result >= INF) ? -1 : result;
}

int main() {
    int n, m;
    cin >> n >> m;

    vector<vector<int>> grid(n, vector<int>(m));
    for (int i = 0; i < n; i++)
        for (int j = 0; j < m; j++)
            cin >> grid[i][j];

    int ans = shortestPath(grid);
    if (ans == -1)
        cout << "No valid path exists\n";
    else
        cout << "Shortest path length: " << ans << "\n";

    return 0;
}
